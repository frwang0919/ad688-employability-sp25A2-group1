from pyspark.ml.clustering import KMeans
from pyspark.ml.evaluation import ClusteringEvaluator
import plotly.graph_objects as go

scores = []
ks = list(range(2, 9))

for k in ks:
    kmeans = KMeans().setK(k).setSeed(42).setFeaturesCol("features")
    model = kmeans.fit(data)
    transformed = model.transform(data)

    evaluator = ClusteringEvaluator(
        featuresCol="features", predictionCol="prediction", metricName="silhouette")
    score = evaluator.evaluate(transformed)
    scores.append(score)

fig = go.Figure()

fig.update_layout(title="Silhouette Score vs. k", 
                  xaxis_title="k",

import numpy as np
import pandas as pd
import plotly.graph_objects as go
from sklearn.mixture import GaussianMixture

# Step 1: Fit GMM on 2D PCA-projected data
X_proj = pdf_gmm[["x", "y"]].values
gmm_model = GaussianMixture(n_components=len(pdf_gmm["prediction"].unique()), covariance_type='full', random_state=0)
gmm_model.fit(X_proj)

# Step 2: Create a mesh grid over x and y
x = np.linspace(X_proj[:, 0].min() - 1, X_proj[:, 0].max() + 1, 300)
y = np.linspace(X_proj[:, 1].min() - 1, X_proj[:, 1].max() + 1, 300)
xx, yy = np.meshgrid(x, y)
grid = np.c_[xx.ravel(), yy.ravel()]

# Step 3: Predict soft probabilities (posterior) over the mesh
probs = gmm_model.predict_proba(grid)
cluster_map = np.argmax(probs, axis=1).reshape(xx.shape)

# Step 4: Plot prediction area + data points
fig = go.Figure()

# Add contour for soft cluster regions
fig.add_trace(go.Contour(
    x=x, y=y, z=cluster_map,
    showscale=False,
    colorscale="Viridis",
    opacity=0.3,
    contours=dict(
        coloring="heatmap",
        showlines=False
    )
))

# Add data points
for label in sorted(pdf_gmm["prediction"].unique()):
    cluster_data = pdf_gmm[pdf_gmm["prediction"] == label]
    fig.add_trace(go.Scatter(
        x=cluster_data["x"], y=cluster_data["y"],
        mode="markers",
        name=f"Cluster {label}",
        marker=dict(size=5),
        hoverinfo="text",
        text=[f"Cluster: {label}" for _ in range(len(cluster_data))]
    ))

# Update layout
fig.update_layout(
    title="GMM Predicted Clusters with Region Overlay (PCA Projection)",
    xaxis_title="PCA 1",
    yaxis_title="PCA 2",
    template="simple_white",
    height=600,
    width=800
)

fig.show()
